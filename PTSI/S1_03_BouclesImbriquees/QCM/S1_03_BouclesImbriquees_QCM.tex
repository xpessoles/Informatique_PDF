def maxi(tab):
   """
   tab est une liste de couples (nom, note)
     * nom est de type str
     * note est un entier entre 0 et 20.
   """
   m = tab[0]
   for x in tab:
      if x[1] >= m[1]:

\item \texttt{[('Grace', 'Hopper', 'F', 1906), ('Ada', 'Lovelace', 'F', 1815)]}.
